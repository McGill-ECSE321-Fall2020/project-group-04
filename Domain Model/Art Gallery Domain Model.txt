class Person
{
  name;
  1 -- 1 Profile profile;    //can a buyer create a profile, or just //artist?
}

class Artist
{
  isA Person;
  1 artist -- * Artwork artworks;
}

class Buyer
{
  isA Person;
}

class Artwork
{
  name;
  year;
  double price;
  boolean isBeingPromoted;
  boolean isSold;
  ArtStyle style;
  ArtworkStatus status
  * artworks -- * Museum museum;
  * interestedArtwork -- * Buyer interestedBuyer;
}

class Museum
{
  double commissionPercentage;
  1 museum <@> 1 MuseumSystem museumSystem;
}

class Profile
{
  username;
  password;
  email;
  double accountBalance;
  AccountType accountType;
  PaymentMethod paymentMethod;
  * profiles <@> 1 MuseumSystem museumSystem;
}

class MuseumSystem 
{
  
}

enum PaymentMethod {CreditCard, PayPal, MORE}
enum ArtworkStatus {Available, Promoted, Sold}
enum ArtStyle {Impressionist, Surrealist, Realist, Renaissance}
enum AccountType {Artist, Buyer};
